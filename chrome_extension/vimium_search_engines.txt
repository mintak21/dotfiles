# 使い方：(q|Q)->g->Space->検索文字列(Googleの場合)

# Google系
g: http://www.google.com/search?q=%s
gt: https://translate.google.com/#en/ja/%s
gm: https://www.google.com/maps?q=%s Google maps
gp: http://www.google.co.jp/search?hl=ja&source=hp&q=%s&tbm=isch Google image
# Qiita
q: https://qiita.com/search?q=%s
qs: https://qiita.com/search?stocked=1&q=%s
# Youtube
y: http://www.youtube.com/results?search_query=%s
# Amazon
a: http://www.amazon.co.jp/s/?field-keywords=%s
